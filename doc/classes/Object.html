def create_nodes(n)
  nodes = (1..n).map{ |i| GraphTest.new i }
  n.times {|i|
    nodes[i].clear_graph
    i.times {|j| nodes[i] << nodes[j] }}
  nodes
end

def pgraph(a, mess)
  puts "\n\n" + mess
  a.each{|g|
    puts "  %s ->" % g
    g.inputs.each{|h|
      puts "    %s" % h
    }
  }
end

def xor(*inp)
  p = 0
  inp.each {|i| p += 1 if i > 0}
  return p == 1
end
